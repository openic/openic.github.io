Introduction

Place "Feedback" files on a server and then adds them to the page. Make sure that the jQuery library is included.
<!-- Add jQuery library -->
<script src="Feedback/scripts/jquery-1.11.0.min.js"></script>

<!-- Add Feedback -->
<script src="Feedback/scripts/html2canvas.js"></script>
<script src="Feedback/scripts/jquery.plugin.html2canvas.js"></script>
<script src="Feedback/scripts/Feedback.language.en.js"></script>
<script src="Feedback/scripts/Feedback.js"></script>

PHP & Cache - configuration

PHP Configure:

This file is in catalog "Feedback/". Before using you must  the configure this file!
<?php

    /**
     * Feedback v1.0
     *
     * Copyright (C) 2014 Erwin Bujak
     * http://lab.erwinbujak.eu/Feedback/
     */

    /**
     * Reporting error
     * Value: E_ALL, E_ERROR, E_WARNING, E_PARSE, E_NOTICE
     */
    ini_set('error_reporting', E_ALL);

    /**
     * Write your subject in email.
     * String
     */
    $mailSubject = 'Feedback from: site name';

    /**
     * Write e-mail address from which the notification will be sent.
     * String
     */
    $mailFromAddress = 'notification@yourdomain.com';

    /**
     * Write your e-mail address to which the notification will come.
     * String
     */
    $mailToAddress = 'example@host.com';

?>

 

Cache configure:

Please change chmod the directory "Feedback / cache /" at the 0777 or 777.

Help how to do it: https://www.youtube.com/watch?v=rda9SpFWL-U

How to use

This is example for using Feedback.

<script>
(function () {
    var config = {
            "url": "Feedback/save.php",
            "multiSelectCategory": false,
            "category": [
                { "name": "General comment", "class": "comment" },
                { "name": "Idea", "class": "idea" },
                { "name": "Error", "class": "error" },
                { "name": "Other" }
            ],
            "fields": [
                {
                    "name": "Message",
                    "type": "textarea",
                    "placeholder": "Write a few words...",
                    "validate": true,
                    "validateMessage": "You must complete a description of the notification."
                },
                {
                    "name": "UrlNotification",
                    "type": "hidden",
                    "value": window.location,
                    "validate": false
                }
            ]
        };

    /* Create button */
    Feedback.button({ name: FeedbackLanguage.buttonInit.name, position: "BottomRightTab" }, config);

})();
</script>
